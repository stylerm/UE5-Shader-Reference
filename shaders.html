Compute Shader Patterns
Compute Shader Entry Point
#include "/Engine/Public/Platform.ush"
#include "/MyPlugin/Private/MyCommon.ush"

// Thread group sizing
#define NUM_THREADS_X 8
#define NUM_THREADS_Y 8

// Resource bindings
Texture2D<float4> InputTexture;
RWTexture2D<float4> OutputTexture;
uint Resolution;

[numthreads(NUM_THREADS_X, NUM_THREADS_Y, 1)]
void MainCS(
    uint3 DispatchThreadID : SV_DispatchThreadID,
    uint3 GroupThreadID : SV_GroupThreadID,
    uint3 GroupID : SV_GroupID)
{
  uint x = DispatchThreadID.x;
  uint y = DispatchThreadID.y;
  if (x >= Resolution || y >= Resolution)
    return;

  // Read, process, write
  float4 Val = InputTexture.Load(
    int3(x, y, 0));
  OutputTexture[uint2(x, y)] = Val;
}
Compute · Common Mistakes
✗Missing bounds check — always guard: if (x >= Res || y >= Res) return;
✗Wrong dispatch size — dispatch = ceil(Resolution / ThreadGroupSize)
✗UAV format mismatch — RWTexture format must match what C++ creates
✗Missing barriers — always GroupMemoryBarrierWithGroupSync() after TGSM writes
✗No ddx/ddy in compute — these are pixel shader only
!Thread group size — 8×8 is common; 16×16 may exceed shared memory limits

Thread Group Shared Memory (TGSM)
TGSM Pattern

Use groupshared memory to share data between threads in a group, reducing global memory reads:

// Declare tile cache
#define TILE_DIM 10  // 8 + 1-pixel halo each side
groupshared float TileCache[TILE_DIM][TILE_DIM];

// 1. Load tile from global memory
int2 LoadPos = GroupID.xy * 8
  + GroupThreadID.xy - 1;  // halo offset
LoadPos = clamp(LoadPos,
  int2(0,0), int2(Res-1,Res-1));
TileCache[GroupThreadID.y][GroupThreadID.x]
  = SourceGrid.Load(int3(LoadPos, 0)).r;

// 2. Sync all threads
GroupMemoryBarrierWithGroupSync();

// 3. Read from cache (fast)
float Center = TileCache[ly][lx];
float Left   = TileCache[ly][lx-1];
float Right  = TileCache[ly][lx+1];
TGSM Rules
✓groupshared keyword declares shared memory — visible to all threads in group
✓Always sync with GroupMemoryBarrierWithGroupSync() between write and read
✓Halo loading — load 1+ pixel border around tile for stencil operations
✓Memory budget — 32KB per thread group on most GPUs; float[10][10] = 400 bytes
!Bank conflicts — stride of 32 floats causes conflicts; pad arrays if needed
!Boundary clamping — halo loads must clamp to grid bounds to avoid OOB reads
TGSM Macro Pattern

Use macros to make TGSM reusable across passes:

// Declare macro for float tiles
#define DECLARE_TILE_CACHE_FLOAT(Name) \
  groupshared float Name[TILE_DIM][TILE_DIM]

// Load macro with scroll compensation
#define LOAD_TILE_SCROLLED(Cache, Tex, ...) \
  { /* load with offset + clamp */ \
    GroupMemoryBarrierWithGroupSync(); }

// Usage in compute shader
DECLARE_TILE_CACHE_FLOAT(SDFCache);
LOAD_TILE_SCROLLED(SDFCache, SDFTex, ...);

Common Shader Errors
!Matrix multiply order — HLSL is row-vector × matrix; GLSL is column-major
!Missing include guard — double-include causes redefinition errors
!Virtual path mismatch — /MyPlugin/ must match AddShaderSourceDirectoryMapping
!Uninitialized groupshared — TGSM is undefined until written
!Integer division — 1/2 = 0; cast to float explicitly: 1.0/2.0
!No implicit broadcast — float3(1) fails; use float3(1,1,1)

Central-Difference Gradient

Compute gradient from a scalar field (e.g. SDF distance):

// Read neighbors
float Left  = Grid.Load(
  int3(max(x-1, 0), y, 0)).r;
float Right = Grid.Load(
  int3(min(x+1, Res-1), y, 0)).r;
float Up    = Grid.Load(
  int3(x, max(y-1, 0), 0)).r;
float Down  = Grid.Load(
  int3(x, min(y+1, Res-1), 0)).r;

// Central difference
float2 Gradient = float2(
  Right - Left, Down - Up);
float Len = length(Gradient);
if (Len > 0.001)
  Gradient /= Len;  // Normalize

// Output: R=dist, GB=direction, A=mag
OutputRT[Pixel] = float4(
  NormDist, Gradient.x, Gradient.y, Len);
SDF Math Helpers
// Smooth minimum (polynomial blend)
float SmoothMin(float a, float b, float k)
{
  if (k <= 0.0) return min(a, b);
  float h = saturate(
    0.5 + 0.5 * (b - a) / k);
  return lerp(b, a, h)
    - k * h * (1.0 - h);
}

// Distance to point
float PointDist(float2 P, float2 C,
    float R)
{
  return length(P - C) - R;
}

// Distance to line segment
float SegmentDist(float2 P,
    float2 A, float2 B, float R)
{
  float2 AB = B - A;
  float t = saturate(
    dot(P - A, AB) / dot(AB, AB));
  return length(P - A - AB * t) - R;
}
